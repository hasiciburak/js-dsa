Arrays

Adding / Removing mutates 
const arr = [1, 2, 3];
arr.push(4);           // [1, 2, 3, 4]
arr.pop();             // [1, 2, 3], returns 4
arr.unshift(0);        // [0, 1, 2, 3]
arr.shift();           // [1, 2, 3], returns 0
arr.splice(1, 1, 'a'); // [1, 'a', 3]
Searching O(n) 
const nums = [10, 20, 30, 20];
nums.indexOf(20);           // 1
nums.lastIndexOf(20);       // 3
nums.includes(30);          // true
nums.find(x => x > 15);     // 20
nums.findIndex(x => x > 15); // 1
Transforming new array 
[1, 2, 3].map(x => x * 2);             // [2, 4, 6]
[1, 2, 3, 4].filter(x => x % 2 === 0); // [2, 4]
[1, 2, 3].reduce((sum, x) => sum + x, 0); // 6
[[1, 2], [3, [4]]].flat(2);         // [1, 2, 3, 4]
Sorting & Slicing O(n log n) 
const unsorted = [3, 1, 4, 1, 5];
unsorted.sort((a, b) => a - b);  // ascending
unsorted.sort((a, b) => b - a);  // descending

const letters = ['a', 'b', 'c', 'd'];
letters.slice(1, 3);   // ['b', 'c']
letters.concat(['e']); // ['a','b','c','d','e']
letters.join('-');     // 'a-b-c-d'

Array.from('abc');                     // ['a','b','c']

DSA Patterns

Stack & Queue O(1) 
// Stack (LIFO)
const stack = [];
stack.push(1);             // [1]
stack.push(2);             // [1, 2]
stack.pop();               // 2, stack is [1]
stack[stack.length - 1];   // peek: 1

// Queue (FIFO) - simple O(n) dequeue
const queue = [];
queue.push(1);             // enqueue
queue.push(2);
queue.shift();             // dequeue: 1
Frequency Counter Map 
function charFrequency(str) {
  const freq = new Map();
  for (const char of str) {
    freq.set(char, (freq.get(char) || 0) + 1);
  }
  return freq;
}
charFrequency('hello');
// Map { 'h'=>1, 'e'=>1, 'l'=>2, 'o'=>1 }
Two Pointers O(n) 
// Palindrome Check
function isPalindrome(s) {
  let left = 0, right = s.length - 1;
  while (left < right) {
    if (s[left] !== s[right]) return false;
    left++;
    right--;
  }
  return true;
}

// Two Sum (sorted array)
function twoSumSorted(arr, target) {
  let left = 0, right = arr.length - 1;
  while (left < right) {
    const sum = arr[left] + arr[right];
    if (sum === target) return [left, right];
    if (sum < target) left++;
    else right--;
  }
  return [-1, -1];
}
Sliding Window O(n) 
// Max Sum of K consecutive elements
function maxSumSubarray(arr, k) {
  let windowSum = 0;
  for (let i = 0; i < k; i++) windowSum += arr[i];
  let maxSum = windowSum;
  for (let i = k; i < arr.length; i++) {
    windowSum += arr[i] - arr[i - k];
    maxSum = Math.max(maxSum, windowSum);
  }
  return maxSum;
}
Binary Search O(log n) 
function binarySearch(arr, target) {
  let lo = 0, hi = arr.length - 1;
  while (lo <= hi) {
    const mid = Math.floor((lo + hi) / 2);
    if (arr[mid] === target) return mid;
    if (arr[mid] < target) lo = mid + 1;
    else hi = mid - 1;
  }
  return -1;
}
Graph Traversal - BFS O(V+E) 
function bfs(graph, start) {
  const visited = new Set([start]);
  const queue = [start];
  while (queue.length > 0) {
    const node = queue.shift();
    for (const neighbor of graph[node] || []) {
      if (!visited.has(neighbor)) {
        visited.add(neighbor);
        queue.push(neighbor);
      }
    }
  }
  return visited;
}
Graph Traversal - DFS O(V+E) 
function dfs(graph, start, visited = new Set()) {
  visited.add(start);
  for (const neighbor of graph[start] || []) {
    if (!visited.has(neighbor)) {
      dfs(graph, neighbor, visited);
    }
  }
  return visited;
}

Quick Reference

Method / Pattern Time Complexity Use Case
arr.push() / pop() O(1) Stack operations
arr.shift() / unshift() O(n) Queue (use linked list for O(1))
arr.sort() O(n log n) Sorting arrays
arr.indexOf() / includes() O(n) Linear search
Binary Search O(log n) Sorted array search
Two Pointers O(n) Pairs, palindromes, sorted arrays
Sliding Window O(n) Subarrays, consecutive elements
Map.get() / set() / has() O(1) avg Frequency counting, lookups
Set.add() / has() / delete() O(1) avg Uniqueness, membership
BFS / DFS O(V + E) Graph traversal
